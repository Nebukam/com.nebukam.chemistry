121 public bool TryGetCandidates(
122 ref NativeList<Neighbor> contents,
123 ref NativeList<int> candidates,
124 ref NativeList<float> weights,
125 out int length)
126 {
127
128 candidates.Clear();
129 weights.Clear();
130
131 if (contents.Length == 0)
132 {
133 length = 0;
134 return false;
135 }
136
137 for (int m = 0; m < m_moduleCount; m++)
138 {
139 if (ModuleMatches(ref contents, ref m))
140 {
141 candidates.Add(m);
142 weights.Add(m_modulesWeights[m]);
143 }
144 }
145
146 length = candidates.Length;
147 return length > 0;
148
149 }

165 {
166
167 contents.Clear();
168
169 int3
170 center = m_inputSlotInfos[index].coord,
171 ccoord;
172
173 int _s;
174
175 // Identify the content of each neighboring socket, if any
176 for (int s = 0; s < m_socketCount; s++)
177 {
178 ccoord = center + m_socketOffsets[s];
179 m_brain.Clamp(ref ccoord);
180
181 if (m_inputSlotCoordinateMap.TryGetValue(ccoord, out _s))
182 _s = m_results[_s];
183 else
184 _s = SlotContent.NULL;
185
186 contents.Add(new Neighbor() { socket = s, value = _s });

320 public bool ModuleMatches(
321 ref ByteTrio coord,
322 ref int moduleIndex)
323 {
324
325 Neighbor content = default;
326 int3
327 center = coord,
328 socket;
329
330 for (int s = 0; s < m_socketCount; s++)
331 {
332
333 content.socket = s;
334
335 socket = center + m_socketOffsets[s];
336 m_brain.Clamp(ref socket);
337
338 if (m_inputSlotCoordinateMap.TryGetValue(socket, out int _s))
339 {
340
341 content.value = m_results[_s];
342
343 if (content.IsUndefined)
344 {
345 if (RequiresNull(ref moduleIndex, ref content.socket))
346 return false;
347 else
348 continue;
349 }
350
351 }
352 else
353 {
354 content.value = SlotContent.NULL;
355 }
356
357 if (!SocketContains(ref moduleIndex, ref content))
358 return false;
359
360 }
361
362 return true;
363
364 }

418 public bool SocketContains(ref int moduleIndex, ref Neighbor content)
419 {
420
421 int3 socketHeader = m_modulesHeaders[moduleIndex * m_socketCount + content.socket];
422
423 for (int i = socketHeader.x; i < socketHeader.y; i++)
424 if (m_modulesNeighbors[i] == content.value)
425 return true;
426
427 return false;
