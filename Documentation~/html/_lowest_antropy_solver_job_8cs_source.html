113 [BurstCompile]
114 public struct SortAntropy : System.Collections.Generic.IComparer<Antropy>
115 {
116 public int Compare(Antropy a, Antropy b)
117 {
118
119 if (a.coord.y < b.coord.y)
120 return 1;
121
122 if (a.candidates - b.candidates < 0)
123 return 1;
124
125 float progress = a.progress - b.progress;
126 return progress > 0f ? 1 : -1;
127 }
128 }

130 public void Execute()
131 {
132
133 #region QueryHandle
134
135 QueryHandler<T_SLOT, T_SLOT_INFOS, T_BRAIN> Q =
136 new QueryHandler<T_SLOT, T_SLOT_INFOS, T_BRAIN>()
137 {
138 m_brain = m_brain,
139 m_inputSlotInfos = m_inputSlotInfos,
140 m_inputSlotCoordinateMap = m_inputSlotCoordinateMap,
141 m_socketCount = m_socketCount,
142 m_socketOffsets = m_socketOffsets,
143 m_socketsMirrors = m_socketsMirrors,
144 m_socketsMirrorsIndices = m_socketsMirrorsIndices,
145 m_moduleCount = m_moduleCount,
146 m_modulesWeights = m_modulesWeights,
147 m_modulesHeaders = m_modulesHeaders,
148 m_modulesNeighbors = m_modulesNeighbors,
149 m_results = m_results,
150 m_nullPairLookup = m_nullPairLookup
151 };
152
153 #endregion
154
155 int
156 sCount = m_inputSlotInfos.Length,
157 nCount = m_modulesNeighbors.Length,
158 count,
159 index,
160 result;
161
162 Antropy _a;
163 SortAntropy antropyComparer = default;
164
165 NativeList<Neighbor>
166 contents = new NativeList<Neighbor>(m_socketCount, Allocator.Temp);
167
168 NativeList<int>
169 candidates = new NativeList<int>(m_modulesNeighbors.Length, Allocator.Temp),
170 slotSocketIndices = new NativeList<int>(m_socketCount, Allocator.Temp),
171 slotIndices = new NativeList<int>(m_socketCount, Allocator.Temp);
172
173 NativeList<float>
174 weights = new NativeList<float>(m_moduleCount, Allocator.Temp);
175
176 NativeList<Antropy>
177 open = new NativeList<Antropy>(sCount, Allocator.Temp);
178
179 #region compute initial antropy
180
181 for (int slotIndex = 0; slotIndex < sCount; slotIndex++)
182 {
183 if (m_results[slotIndex] < 0 &&
184 Q.TryGetCandidates(
185 slotIndex,
186 ref contents,
187 ref candidates,
188 ref weights,
189 out count))
190 {
191 _a = new Antropy()
192 {
193 coord = m_inputSlotInfos[slotIndex].coord,
194 index = slotIndex,
195 candidates = count,
196 progress = Q.GetNeighborsResolutionRatio(slotIndex)
197 };
198
199 if (open.Length != 0)
200 {
201 if (antropyComparer.Compare(_a, open[0]) >= 0)
202 open[0] = _a;
203 }
204 else
205 {
206 open.Add(_a);
207 }
208
209 }
210 }
211
212 if (open.Length != 0)
213 open.Sort(antropyComparer);
214 else
215 open.Add(new Antropy() { index = 0, candidates = m_moduleCount });
216
217 // ISSUE : When resolving candidates, some header gets selected because a neighboring socket is UNSET, and thus ignored
218 // while the header only accepts NULL for that socket.
219
220 #endregion
221
222 #region collapse
223
224 int increment = 0;
225 float ratio;
226
227 while (open.Length > 0)
228 {
229 _a = open.Pop();
230 index = _a.index;
231 m_debug[index] = increment / (float)sCount;
232
233 // Collapse
234
235 if (Q.TryGetCandidates(
236 index,
237 ref contents,
238 ref candidates,
239 ref weights,
240 out count))
241 {
242 result = candidates[ NRandom.GetRandomWeightedIndex(ref weights, NextFloat()) ];
243 }
244 else
245 {
246 // Uh oh, no candidate found !
247 result = SlotContent.UNSOLVABLE;
248 }
249
250 m_results[index] = result;
251
252 // Recalculate antropy of neighboring slots that are currently unset
253
254 if (Q.TryGetMatchingNeighbors(
255 index, SlotContent.UNSET,
256 ref slotSocketIndices,
257 ref slotIndices,
258 out count))
259 {
260 for (int s = 0; s < count; s++)
261 {
262 index = slotIndices[s];
263 if (Q.TryGetCandidates(
264 index,
265 ref contents,
266 ref candidates,
267 ref weights,
268 out int antropy))
269 {
270 ratio = Q.GetNeighborsResolutionRatio(index);
271 UpdateAntropy(ref index, ref antropy, ref ratio, ref open);
272 }
273 }
274 }
275
276 // Sort open slots which antropy is computed
277
278 open.Sort(antropyComparer);
279 increment++;
280 }
281
282 #endregion
283
284 open.Dispose();
285 contents.Dispose();
286 candidates.Dispose();
287 slotSocketIndices.Dispose();
288 slotIndices.Dispose();
289 weights.Dispose();
290 }

292 private void UpdateAntropy(
293 ref int index,
294 ref int antropy,
295 ref float ratio,
296 ref NativeList<Antropy> open)
297 {
298
299 Antropy a;
300 for (int i = 0, count = open.Length; i < count; i++)
301 {
302 a = open[i];
303 if (a.index == index)
304 {
305 a.candidates = antropy;
306 a.progress = ratio;
307 open[i] = a;
308 return;
309 }
310 }
311
312 open.Add(new Antropy()
313 {
314 coord = m_inputSlotInfos[index].coord,
315 index = index,
316 candidates = antropy,
317 progress = ratio
318 });
319
320 }
