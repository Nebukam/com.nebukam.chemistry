103 public void Execute()
104 {
105
106 #region QueryHandle
107
108 QueryHandler<T_SLOT, T_SLOT_INFOS, T_BRAIN> Q =
109 new QueryHandler<T_SLOT, T_SLOT_INFOS, T_BRAIN>()
110 {
111 m_brain = m_brain,
112 m_inputSlotInfos = m_inputSlotInfos,
113 m_inputSlotCoordinateMap = m_inputSlotCoordinateMap,
114 m_socketCount = m_socketCount,
115 m_socketOffsets = m_socketOffsets,
116 m_socketsMirrors = m_socketsMirrors,
117 m_socketsMirrorsIndices = m_socketsMirrorsIndices,
118 m_moduleCount = m_moduleCount,
119 m_modulesWeights = m_modulesWeights,
120 m_modulesHeaders = m_modulesHeaders,
121 m_modulesNeighbors = m_modulesNeighbors,
122 m_results = m_results,
123 m_nullPairLookup = m_nullPairLookup
124 };
125
126 #endregion
127
128 NativeList<Neighbor>
129 contents = new NativeList<Neighbor>(m_socketCount, Allocator.Temp);
130
131 NativeList<int>
132 candidates = new NativeList<int>(m_modulesNeighbors.Length, Allocator.Temp),
133 unsolvables = new NativeList<int>(10, Allocator.Temp);
134
135 NativeList<float>
136 weights = new NativeList<float>(m_moduleCount, Allocator.Temp);
137
138 int
139 cCount,
140 result;
141
142 for (int slotIndex = 0, count = m_inputSlotInfos.Length; slotIndex < count; slotIndex++)
143 {
144
145 if (m_results[slotIndex] >= 0)
146 continue;
147
148 if (Q.TryGetCandidates(
149 slotIndex,
150 ref contents,
151 ref candidates,
152 ref weights,
153 out cCount))
154 {
155 result = candidates[NRandom.GetRandomWeightedIndex(ref weights, NextFloat())];
156 }
157 else
158 {
159 result = SlotContent.UNSOLVABLE;
160 unsolvables.Add(slotIndex);
161 }
162
163 m_results[slotIndex] = result;
164
165 }
166
167 // Second pass if necessary
168
169 if (unsolvables.Length != 0)
170 {
171 int index = 0;
172
173 for (int u = 0, uCount = unsolvables.Length; u < uCount; u++)
174 {
175 index = unsolvables[u];
176
177 if (Q.TryGetCandidates(
178 index,
179 ref contents,
180 ref candidates,
181 ref weights,
182 out cCount))
183 {
184 result = candidates[NRandom.GetRandomWeightedIndex(ref weights, NextFloat())];
185 }
186 else
187 {
188 result = SlotContent.UNSOLVABLE;
189 }
190
191 m_results[index] = result;
192 }
193 }
194
195 contents.Dispose();
196 candidates.Dispose();
197 unsolvables.Dispose();
198 weights.Dispose();
199
200 }
